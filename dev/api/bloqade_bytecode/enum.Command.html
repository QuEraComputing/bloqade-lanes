Enum Command

pub(crate) enum Command {
    Assemble {
        input: PathBuf,
        output: PathBuf,
    },
    Disassemble {
        input: PathBuf,
        output: Option<PathBuf>,
    },
    Validate {
        input: PathBuf,
        arch: Option<PathBuf>,
        simulate_stack: bool,
    },
    Arch {
        input: Option<PathBuf>,
        command: Option<ArchCommand>,
    },
    EvalPolicy {
        policy: PathBuf,
        problem: PathBuf,
        params: Option<PathBuf>,
        max_expansions: Option<u64>,
        timeout: Option<f64>,
        json: bool,
        seed: Option<u64>,
    },
    TracePolicy {
        policy: PathBuf,
        problem: PathBuf,
        params: Option<PathBuf>,
        max_expansions: Option<u64>,
        timeout: Option<f64>,
        json: bool,
        seed: Option<u64>,
        out: Option<PathBuf>,
    },
}
