1//! Structural validation for [`ArchSpec`].
2//!
3//! Validates all structural rules in a single pass, collecting every error
4//! rather than failing fast. See [`ArchSpec::validate`].
5
6use std::collections::HashSet;
7
8use thiserror::Error;
9
10use super::types::ArchSpec;
11
12/// Error categories for arch spec structural validation.
13///
14/// Each variant groups related validation checks. Multiple errors
15/// can be collected in a single validation pass.
16#[derive(Debug, Clone, PartialEq, Error)]
17pub enum ArchSpecError {
18    /// Structural error (grid dimensions, word consistency, minimum counts).
19    #[error("{0}")]
20    Structure(String),
21
22    /// Per-zone bus validation error (site/word buses, membership lists).
23    #[error("{0}")]
24    ZoneBus(String),
25
26    /// Inter-zone bus validation error (zone bus entries).
27    #[error("{0}")]
28    InterZoneBus(String),
29
30    /// Grid invariant error (bus src/dst not rectangular).
31    #[error("{0}")]
32    GridInvariant(String),
33
34    /// Entangling zone pair validation error.
35    #[error("{0}")]
36    EntanglingPair(String),
37
38    /// Mode validation error.
39    #[error("{0}")]
40    Mode(String),
41
42    /// Transport path validation error.
43    #[error("{0}")]
44    Path(String),
45}
46
47impl ArchSpec {
48    /// Validate the arch spec against all structural rules.
49    /// Collects all errors in one pass (not fail-fast).
50    pub fn validate(&self) -> Result<(), Vec<ArchSpecError>> {
51        let mut errors = Vec::new();
52
53        let num_words = self.words.len();
54        let num_zones = self.zones.len();
55        let sites_per_word = self.sites_per_word();
56
57        // Structural invariants
58        check_minimum_counts(self, &mut errors);
59        check_uniform_grid_dimensions(self, &mut errors);
60        check_uniform_word_site_counts(self, &mut errors);
61        check_word_site_indices(self, &mut errors);
62
63        // Per-zone bus and entangling pair validation
64        for (zone_idx, zone) in self.zones.iter().enumerate() {
65            check_zone_words_with_site_buses(zone_idx, zone, num_words, &mut errors);
66            check_zone_sites_with_word_buses(zone_idx, zone, sites_per_word, &mut errors);
67            check_zone_site_buses(zone_idx, zone, sites_per_word, &mut errors);
68            check_zone_word_buses(zone_idx, zone, num_words, &mut errors);
69            check_zone_entangling_pairs(zone_idx, zone, num_words, &mut errors);
70        }
71
72        // Inter-zone bus validation
73        check_zone_buses(self, num_zones, num_words, &mut errors);
74
75        // Mode validation
76        check_modes(self, num_zones, num_words, sites_per_word, &mut errors);
77
78        // Path validation
79        check_paths(self, num_zones, &mut errors);
80
81        if errors.is_empty() {
82            Ok(())
83        } else {
84            Err(errors)
85        }
86    }
87}

89/// At least one zone and one word must exist.
90fn check_minimum_counts(spec: &ArchSpec, errors: &mut Vec<ArchSpecError>) {
91    if spec.zones.is_empty() {
92        errors.push(ArchSpecError::Structure(
93            "at least one zone must exist".into(),
94        ));
95    }
96    if spec.words.is_empty() {
97        errors.push(ArchSpecError::Structure(
98            "at least one word must exist".into(),
99        ));
100    }
101}

103/// All zones must have the same grid dimensions (same num_x and num_y).
104fn check_uniform_grid_dimensions(spec: &ArchSpec, errors: &mut Vec<ArchSpecError>) {
105    if let Some(first_zone) = spec.zones.first() {
106        let ref_x = first_zone.grid.num_x();
107        let ref_y = first_zone.grid.num_y();
108        for (idx, zone) in spec.zones.iter().enumerate().skip(1) {
109            let zx = zone.grid.num_x();
110            let zy = zone.grid.num_y();
111            if zx != ref_x || zy != ref_y {
112                errors.push(ArchSpecError::Structure(format!(
113                    "zone {} grid dimensions ({}x{}) differ from zone 0 ({}x{})",
114                    idx, zx, zy, ref_x, ref_y
115                )));
116            }
117        }
118    }
119}
120
121/// All words must have the same number of sites.
122fn check_uniform_word_site_counts(spec: &ArchSpec, errors: &mut Vec<ArchSpecError>) {
123    if let Some(first_word) = spec.words.first() {
124        let ref_count = first_word.sites.len();
125        for (idx, word) in spec.words.iter().enumerate().skip(1) {
126            if word.sites.len() != ref_count {
127                errors.push(ArchSpecError::Structure(format!(
128                    "word {} has {} sites, expected {} (same as word 0)",
129                    idx,
130                    word.sites.len(),
131                    ref_count
132                )));
133            }
134        }
135    }
136}

138/// Word site indices must be within the grid dimensions of every zone.
139/// For each site [x, y]: x < grid.num_x() and y < grid.num_y().
140fn check_word_site_indices(spec: &ArchSpec, errors: &mut Vec<ArchSpecError>) {
141    // Use zone 0's grid as the reference (uniform dimensions already checked).
142    let (grid_x, grid_y) = match spec.zones.first() {
143        Some(z) => (z.grid.num_x(), z.grid.num_y()),
144        None => return, // no zones → nothing to check
145    };
146
147    for (word_idx, word) in spec.words.iter().enumerate() {
148        for (site_idx, site) in word.sites.iter().enumerate() {
149            let x = site[0] as usize;
150            let y = site[1] as usize;
151            if x >= grid_x {
152                errors.push(ArchSpecError::Structure(format!(
153                    "word {}, site {}: x index {} out of range (grid has {} x-positions)",
154                    word_idx, site_idx, site[0], grid_x
155                )));
156            }
157            if y >= grid_y {
158                errors.push(ArchSpecError::Structure(format!(
159                    "word {}, site {}: y index {} out of range (grid has {} y-positions)",
160                    word_idx, site_idx, site[1], grid_y
161                )));
162            }
163        }
164    }
165}

171/// `words_with_site_buses` entries must be < number of words.
172fn check_zone_words_with_site_buses(
173    zone_idx: usize,
174    zone: &Zone,
175    num_words: usize,
176    errors: &mut Vec<ArchSpecError>,
177) {
178    for &wid in &zone.words_with_site_buses {
179        if wid as usize >= num_words {
180            errors.push(ArchSpecError::ZoneBus(format!(
181                "zone {}: words_with_site_buses contains invalid word ID {}",
182                zone_idx, wid
183            )));
184        }
185    }
186}

188/// `sites_with_word_buses` entries must be valid site indices.
189fn check_zone_sites_with_word_buses(
190    zone_idx: usize,
191    zone: &Zone,
192    sites_per_word: usize,
193    errors: &mut Vec<ArchSpecError>,
194) {
195    for &sid in &zone.sites_with_word_buses {
196        if sid as usize >= sites_per_word {
197            errors.push(ArchSpecError::ZoneBus(format!(
198                "zone {}: sites_with_word_buses contains invalid site index {} (sites_per_word={})",
199                zone_idx, sid, sites_per_word
200            )));
201        }
202    }
203}

205/// Site bus src/dst must have same length and SiteRef values < sites_per_word.
206fn check_zone_site_buses(
207    zone_idx: usize,
208    zone: &Zone,
209    sites_per_word: usize,
210    errors: &mut Vec<ArchSpecError>,
211) {
212    for (bus_idx, bus) in zone.site_buses.iter().enumerate() {
213        if bus.src.len() != bus.dst.len() {
214            errors.push(ArchSpecError::ZoneBus(format!(
215                "zone {}, site_bus {}: src length ({}) != dst length ({})",
216                zone_idx,
217                bus_idx,
218                bus.src.len(),
219                bus.dst.len()
220            )));
221        }
222        for (i, sref) in bus.src.iter().enumerate() {
223            if sref.0 as usize >= sites_per_word {
224                errors.push(ArchSpecError::ZoneBus(format!(
225                    "zone {}, site_bus {}: src[{}] SiteRef({}) >= sites_per_word ({})",
226                    zone_idx, bus_idx, i, sref.0, sites_per_word
227                )));
228            }
229        }
230        for (i, sref) in bus.dst.iter().enumerate() {
231            if sref.0 as usize >= sites_per_word {
232                errors.push(ArchSpecError::ZoneBus(format!(
233                    "zone {}, site_bus {}: dst[{}] SiteRef({}) >= sites_per_word ({})",
234                    zone_idx, bus_idx, i, sref.0, sites_per_word
235                )));
236            }
237        }
238    }
239}

241/// Word bus src/dst must have same length and WordRef values < number of words.
242fn check_zone_word_buses(
243    zone_idx: usize,
244    zone: &Zone,
245    num_words: usize,
246    errors: &mut Vec<ArchSpecError>,
247) {
248    for (bus_idx, bus) in zone.word_buses.iter().enumerate() {
249        if bus.src.len() != bus.dst.len() {
250            errors.push(ArchSpecError::ZoneBus(format!(
251                "zone {}, word_bus {}: src length ({}) != dst length ({})",
252                zone_idx,
253                bus_idx,
254                bus.src.len(),
255                bus.dst.len()
256            )));
257        }
258        for (i, wref) in bus.src.iter().enumerate() {
259            if wref.0 as usize >= num_words {
260                errors.push(ArchSpecError::ZoneBus(format!(
261                    "zone {}, word_bus {}: src[{}] WordRef({}) >= num_words ({})",
262                    zone_idx, bus_idx, i, wref.0, num_words
263                )));
264            }
265        }
266        for (i, wref) in bus.dst.iter().enumerate() {
267            if wref.0 as usize >= num_words {
268                errors.push(ArchSpecError::ZoneBus(format!(
269                    "zone {}, word_bus {}: dst[{}] WordRef({}) >= num_words ({})",
270                    zone_idx, bus_idx, i, wref.0, num_words
271                )));
272            }
273        }
274    }
275}

277// --- Inter-zone bus validation ---
278
279/// Zone bus entries must have valid zone_id and word_id, src/dst same length,
280/// and every pair must cross a zone boundary.
281fn check_zone_buses(
282    spec: &ArchSpec,
283    num_zones: usize,
284    num_words: usize,
285    errors: &mut Vec<ArchSpecError>,
286) {
287    for (bus_idx, bus) in spec.zone_buses.iter().enumerate() {
288        if bus.src.len() != bus.dst.len() {
289            errors.push(ArchSpecError::InterZoneBus(format!(
290                "zone_bus {}: src length ({}) != dst length ({})",
291                bus_idx,
292                bus.src.len(),
293                bus.dst.len()
294            )));
295        }
296
297        // Validate all ZonedWordRef entries
298        for (i, zwr) in bus.src.iter().enumerate() {
299            if zwr.zone_id as usize >= num_zones {
300                errors.push(ArchSpecError::InterZoneBus(format!(
301                    "zone_bus {}: src[{}] zone_id {} >= num_zones ({})",
302                    bus_idx, i, zwr.zone_id, num_zones
303                )));
304            }
305            if zwr.word_id as usize >= num_words {
306                errors.push(ArchSpecError::InterZoneBus(format!(
307                    "zone_bus {}: src[{}] word_id {} >= num_words ({})",
308                    bus_idx, i, zwr.word_id, num_words
309                )));
310            }
311        }
312        for (i, zwr) in bus.dst.iter().enumerate() {
313            if zwr.zone_id as usize >= num_zones {
314                errors.push(ArchSpecError::InterZoneBus(format!(
315                    "zone_bus {}: dst[{}] zone_id {} >= num_zones ({})",
316                    bus_idx, i, zwr.zone_id, num_zones
317                )));
318            }
319            if zwr.word_id as usize >= num_words {
320                errors.push(ArchSpecError::InterZoneBus(format!(
321                    "zone_bus {}: dst[{}] word_id {} >= num_words ({})",
322                    bus_idx, i, zwr.word_id, num_words
323                )));
324            }
325        }
326
327        // Every (src[i], dst[i]) pair must cross a zone boundary
328        let pair_count = bus.src.len().min(bus.dst.len());
329        for i in 0..pair_count {
330            if bus.src[i].zone_id == bus.dst[i].zone_id {
331                errors.push(ArchSpecError::InterZoneBus(format!(
332                    "zone_bus {}: pair {} does not cross a zone boundary \
333                     (src zone_id={}, dst zone_id={})",
334                    bus_idx, i, bus.src[i].zone_id, bus.dst[i].zone_id
335                )));
336            }
337        }
338    }
339}

341// --- Per-zone entangling pair validation ---
342
343/// Word indices in entangling_pairs must be valid, distinct, and no duplicate pairs.
344fn check_zone_entangling_pairs(
345    zone_idx: usize,
346    zone: &super::types::Zone,
347    num_words: usize,
348    errors: &mut Vec<ArchSpecError>,
349) {
350    let mut seen: HashSet<[u32; 2]> = HashSet::new();
351    for (idx, pair) in zone.entangling_pairs.iter().enumerate() {
352        let [a, b] = *pair;
353        if a as usize >= num_words {
354            errors.push(ArchSpecError::EntanglingPair(format!(
355                "zone[{}].entangling_pairs[{}]: word ID {} >= num_words ({})",
356                zone_idx, idx, a, num_words
357            )));
358        }
359        if b as usize >= num_words {
360            errors.push(ArchSpecError::EntanglingPair(format!(
361                "zone[{}].entangling_pairs[{}]: word ID {} >= num_words ({})",
362                zone_idx, idx, b, num_words
363            )));
364        }
365        if a == b {
366            errors.push(ArchSpecError::EntanglingPair(format!(
367                "zone[{}].entangling_pairs[{}]: word paired with itself ({})",
368                zone_idx, idx, a
369            )));
370        }
371        let normalized = if a <= b { [a, b] } else { [b, a] };
372        if !seen.insert(normalized) {
373            errors.push(ArchSpecError::EntanglingPair(format!(
374                "zone[{}].entangling_pairs[{}]: duplicate pair [{}, {}]",
375                zone_idx, idx, a, b
376            )));
377        }
378    }
379}

381// --- Mode validation ---
382
383/// Zone indices and bitstring_order entries must be valid.
384fn check_modes(
385    spec: &ArchSpec,
386    num_zones: usize,
387    num_words: usize,
388    sites_per_word: usize,
389    errors: &mut Vec<ArchSpecError>,
390) {
391    for (mode_idx, mode) in spec.modes.iter().enumerate() {
392        for &zone_id in &mode.zones {
393            if zone_id as usize >= num_zones {
394                errors.push(ArchSpecError::Mode(format!(
395                    "mode '{}' (index {}): zone ID {} >= num_zones ({})",
396                    mode.name, mode_idx, zone_id, num_zones
397                )));
398            }
399        }
400        for (loc_idx, loc) in mode.bitstring_order.iter().enumerate() {
401            if loc.zone_id as usize >= num_zones {
402                errors.push(ArchSpecError::Mode(format!(
403                    "mode '{}' (index {}): bitstring_order[{}] zone_id {} >= num_zones ({})",
404                    mode.name, mode_idx, loc_idx, loc.zone_id, num_zones
405                )));
406            }
407            if loc.word_id as usize >= num_words {
408                errors.push(ArchSpecError::Mode(format!(
409                    "mode '{}' (index {}): bitstring_order[{}] word_id {} >= num_words ({})",
410                    mode.name, mode_idx, loc_idx, loc.word_id, num_words
411                )));
412            }
413            if loc.site_id as usize >= sites_per_word {
414                errors.push(ArchSpecError::Mode(format!(
415                    "mode '{}' (index {}): bitstring_order[{}] site_id {} >= sites_per_word ({})",
416                    mode.name, mode_idx, loc_idx, loc.site_id, sites_per_word
417                )));
418            }
419        }
420    }
421}

423// --- Path validation ---
424
425/// If paths is Some, validate each path's waypoints and lane address.
426fn check_paths(spec: &ArchSpec, num_zones: usize, errors: &mut Vec<ArchSpecError>) {
427    if let Some(paths) = &spec.paths {
428        for (idx, path) in paths.iter().enumerate() {
429            if !path.check_finite() {
430                errors.push(ArchSpecError::Path(format!(
431                    "paths[{}]: waypoint contains non-finite coordinate",
432                    idx
433                )));
434            }
435
436            // Validate the lane address fields
437            let lane = super::addr::LaneAddr::decode_u64(path.lane);
438            if lane.zone_id as usize >= num_zones {
439                errors.push(ArchSpecError::Path(format!(
440                    "paths[{}]: lane 0x{:016X} has invalid zone_id {} (num_zones={})",
441                    idx, path.lane, lane.zone_id, num_zones
442                )));
443            }
444
445            if path.waypoints.len() < 2 {
446                errors.push(ArchSpecError::Path(format!(
447                    "paths[{}]: lane 0x{:016X} has {} waypoint(s), minimum is 2",
448                    idx,
449                    path.lane,
450                    path.waypoints.len()
451                )));
452            }
453        }
454    }
455}
