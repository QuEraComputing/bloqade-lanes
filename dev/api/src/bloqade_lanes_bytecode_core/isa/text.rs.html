1//! Text (`.sst`) codec for the vihaco-backed ISA — vihaco `ParsedModule` grammar.
2//!
3//! The grammar is:
4//! ```text
5//! version <major>.<minor>;
6//! fn @main() {
7//!   <instruction>
8//!   ...
9//! }
10//! ```
11//!
12//! `LanesHeader` parses the `version <major>.<minor>` directive.
13//! `LanesResolver` lowers a `ParsedModule<Instruction, LanesHeader>` into a
14//! `Program` via [`super::program::from_code`].
15
16use chumsky::prelude::*;
17use vihaco::syntax::{ParsedModule, Resolve};
18use vihaco_parser_core::Parse;
19
20use super::Instruction;
21use super::program::{Program, from_code};
22use crate::version::Version;

24/// Error from text (`.sst`) parsing.
25#[derive(Debug, Clone, PartialEq, Eq)]
26pub enum TextError {
27    /// No `version M.N;` header was found before the first instruction.
28    MissingVersion,
29    /// The version header's value could not be parsed.
30    /// Currently unreachable: a malformed version fails the whole parse via `BadInstruction`.
31    /// Retained for API stability and potential future use.
32    InvalidVersion { line: usize, value: String },
33    /// A line could not be parsed as an instruction. `line` is currently always
34    /// `0` for parse-level failures via the vihaco `ParsedModule` parser.
35    BadInstruction { line: usize, text: String },
36}
37
38impl std::fmt::Display for TextError {
39    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
40        match self {
41            TextError::MissingVersion => write!(f, "missing version header"),
42            TextError::InvalidVersion { line, value } => {
43                write!(f, "line {line}: invalid version '{value}'")
44            }
45            TextError::BadInstruction { line, text } => {
46                write!(f, "line {line}: cannot parse instruction '{text}'")
47            }
48        }
49    }
50}
51
52impl std::error::Error for TextError {}

56/// The only header we support: `version <major>.<minor>`.
57#[derive(Debug, Clone, PartialEq)]
58pub enum LanesHeader {
59    Version(Version),
60}
61
62impl<'src> Parse<'src> for LanesHeader {
63    fn parser() -> impl chumsky::Parser<
64        'src,
65        &'src str,
66        Self,
67        chumsky::extra::Err<chumsky::error::Simple<'src, char>>,
68    > {
69        let uint = || {
70            any()
71                .filter(|c: &char| c.is_ascii_digit())
72                .repeated()
73                .at_least(1)
74                .collect::<String>()
75        };
76        just("version")
77            .ignore_then(chumsky::text::whitespace())
78            .ignore_then(uint().then_ignore(just('.')).then(uint()))
79            .try_map(|(maj, min), span| {
80                let major = maj
81                    .parse::<u16>()
82                    .map_err(|_| chumsky::error::Simple::new(None, span))?;
83                let minor = min
84                    .parse::<u16>()
85                    .map_err(|_| chumsky::error::Simple::new(None, span))?;
86                Ok(LanesHeader::Version(Version::new(major, minor)))
87            })
88    }
89}

91// ── LanesResolver ─────────────────────────────────────────────────────────────
92
93struct LanesResolver;
94
95impl Resolve<Instruction, LanesHeader> for LanesResolver {
96    type Module = Program;
97
98    fn resolve_module(
99        &mut self,
100        parsed: ParsedModule<Instruction, LanesHeader>,
101    ) -> eyre::Result<Program> {
102        // Exactly one `version M.N;` header. Additional headers almost always
103        // signal a mistake (conflicting versions), so reject them rather than
104        // silently taking the first.
105        let version = match parsed.headers.as_slice() {
106            [] => eyre::bail!("missing version header"),
107            [LanesHeader::Version(v)] => *v,
108            _ => eyre::bail!("multiple version headers"),
109        };
110
111        // Exactly one function, and it must be `@main`. The parser strips the
112        // leading `@`, so the resolved name is bare `"main"`.
113        let func = match parsed.functions.as_slice() {
114            [f] => f,
115            _ => eyre::bail!("expected exactly one function (@main)"),
116        };
117        if func.name != "main" {
118            eyre::bail!("expected function @main, found @{}", func.name);
119        }
120
121        // Default resolve_body: all lanes instructions are Direct; Raw is an error.
122        let code = self.resolve_body(func.body.clone())?;
123        Ok(from_code(version, code))
124    }
125}

127// ── map_resolve_err ───────────────────────────────────────────────────────────
128
129fn map_resolve_err(err: &eyre::Report) -> TextError {
130    let msg = err.to_string();
131    if msg.contains("missing version header") {
132        TextError::MissingVersion
133    } else {
134        TextError::BadInstruction { line: 0, text: msg }
135    }
136}

140/// Parse the vihaco `fn @main` grammar into a [`Program`].
141///
142/// Expected format:
143/// ```text
144/// version <major>.<minor>;
145/// fn @main() {
146///   <instruction>
147///   ...
148/// }
149/// ```
150///
151/// Note: `TextError::BadInstruction` errors have `line: 0` for parse-level
152/// failures via vihaco `ParsedModule`; the underlying chumsky diagnostics are
153/// preserved in `text` even though a precise line number is not yet available.
154pub fn parse_text(src: &str) -> Result<Program, TextError> {
155    let parsed = ParsedModule::<Instruction, LanesHeader>::parser()
156        .parse(src)
157        .into_result()
158        .map_err(|errs| TextError::BadInstruction {
159            line: 0,
160            text: errs
161                .iter()
162                .map(|e| e.to_string())
163                .collect::<Vec<_>>()
164                .join("; "),
165        })?;
166    LanesResolver
167        .resolve_module(parsed)
168        .map_err(|e| map_resolve_err(&e))
169}
170
171/// Emit the program as the vihaco `fn @main` grammar.
172///
173/// The output is accepted by [`parse_text`] and round-trips losslessly.
174pub fn to_text(program: &Program) -> String {
175    let mut out = format!(
176        "version {}.{};\nfn @main() {{\n",
177        program.extra.version.major, program.extra.version.minor
178    );
179    for inst in &program.code {
180        out.push_str("  ");
181        out.push_str(&inst.to_string());
182        out.push('\n');
183    }
184    out.push_str("}\n");
185    out
186}

225        let src = "version 1.2;\nfn @main() {\n  const_loc 0x0000000000000000\n  initial_fill 1\n  halt\n}\n";
226        let p = parse_text(src).unwrap();
227        assert_eq!(p.extra.version, Version::new(1, 2));
228        assert_eq!(p.code.len(), 3);
